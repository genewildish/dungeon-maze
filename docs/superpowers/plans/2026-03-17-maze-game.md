# Maze Game Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-based maze game with squiggle-method generation, arrow-key navigation, optional trail/checkpoint toggles, and retro pixel art sparkle effects — all in a single `index.html` file.

**Architecture:** Single HTML file with inline CSS and JS. The JS is organized into logical sections: grid/maze generation, rendering, player/input, trail, checkpoint, particle system, and game state/screens. Each section is a set of functions operating on shared state objects (grid, player, particles array). A `requestAnimationFrame` game loop drives rendering and particle updates.

**Tech Stack:** HTML5 Canvas, Vanilla JavaScript, CSS (no dependencies)

---

## File Structure

```
maze/
  index.html    — the entire game
```

Single file. HTML structure at top (canvas, overlay divs for start/dead-end/win screens), CSS in a `<style>` block, JS in a `<script>` block organized into these sections:

2. **Grid & Maze Generation** — `createGrid()`, `generateSolutionPath()`, `fillMaze()`
3. **Rendering** — `render()`, `drawMaze()`, `drawPlayer()`, `drawTrail()`
4. **Player & Input** — `movePlayer()`, `keydown` handler, wall collision
5. **Trail** — Visited cell tracking, rendering
6. **Checkpoint** — Solution path lookup, `lastCheckpoint` update, `lastCheckpointIndex`
7. **Dead End Detection** — `isDeadEnd()`, manual help trigger
8. **Particle System** — `spawnParticles()`, `updateParticles()`, `drawParticles()`
9. **Screens** — Start screen, dead-end modal, win screen
10. **Game Loop** — `requestAnimationFrame` loop, state machine (start → playing → dead-end → win)

```javascript
    // === GRID & MAZE GENERATION ===

    function createGrid() {
      grid = [];
      for (let y = 0; y < ROWS; y++) {
        grid[y] = [];
        for (let x = 0; x < COLS; x++) {
          grid[y][x] = {
            walls: { n: true, s: true, e: true, w: true },
            visited: false
          };
        }
      }
    }

    function removeWall(x1, y1, x2, y2) {
      if (x2 === x1 + 1) { grid[y1][x1].walls.e = false; grid[y2][x2].walls.w = false; }
      if (x2 === x1 - 1) { grid[y1][x1].walls.w = false; grid[y2][x2].walls.e = false; }
      if (y2 === y1 + 1) { grid[y1][x1].walls.s = false; grid[y2][x2].walls.n = false; }
      if (y2 === y1 - 1) { grid[y1][x1].walls.n = false; grid[y2][x2].walls.s = false; }
    }

    function getNeighbors(x, y, onlyUnvisited) {
      const neighbors = [];
      if (y > 0 && (!onlyUnvisited || !grid[y - 1][x].visited)) neighbors.push([x, y - 1, 'n']);
      if (y < ROWS - 1 && (!onlyUnvisited || !grid[y + 1][x].visited)) neighbors.push([x, y + 1, 's']);
      if (x > 0 && (!onlyUnvisited || !grid[y][x - 1].visited)) neighbors.push([x - 1, y, 'w']);
      if (x < COLS - 1 && (!onlyUnvisited || !grid[y][x + 1].visited)) neighbors.push([x + 1, y, 'e']);
      return neighbors;
    }
```

### Task 3: Phase 1 — Solution Path Generation

- [ ] **Step 1: Add solution path generator**

```javascript
    function generateSolutionPath() {
      solutionPath = [];
      const startY = Math.floor(Math.random() * ROWS);
      let x = 0, y = startY;

      grid[y][x].visited = true;
      solutionPath.push([x, y]);

      // Open entry wall
      grid[y][x].walls.w = false;

      while (x < COLS - 1) {
        const neighbors = getNeighbors(x, y, true); // only unvisited
        if (neighbors.length === 0) {
          // Backtrack along path to find cell with unvisited neighbors
          solutionPath.pop();
          if (solutionPath.length === 0) break; // shouldn't happen on 25x25
          const prev = solutionPath[solutionPath.length - 1];
          x = prev[0];
          y = prev[1];
          continue;
        }

        // Weighted direction selection: E=40%, N=20%, S=20%, W=20%
        const weighted = [];
        for (const [nx, ny, dir] of neighbors) {
          const weight = dir === 'e' ? 40 : 20;
          for (let i = 0; i < weight; i++) weighted.push([nx, ny]);
        }
        const [nx, ny] = weighted[Math.floor(Math.random() * weighted.length)];

        removeWall(x, y, nx, ny);
        grid[ny][nx].visited = true;
        solutionPath.push([nx, ny]);
        x = nx;
        y = ny;
      }

      // Open exit wall
      const lastCell = solutionPath[solutionPath.length - 1];
      grid[lastCell[1]][lastCell[0]].walls.e = false;
    }
```

### Task 4: Phase 2 — Recursive Backtracker (Fill the Rest)

**Files:**
- Modify: `index.html` (JS section, after solution path)

- [ ] **Step 1: Add recursive backtracker**

```javascript
    function fillMaze() {
      // Phase 2: recursive backtracker from each unvisited cell
      for (let y = 0; y < ROWS; y++) {
        for (let x = 0; x < COLS; x++) {
          if (!grid[y][x].visited) {
            carve(x, y);
          }
        }
      }
    }

    function carve(startX, startY) {
      const stack = [[startX, startY]];
      grid[startY][startX].visited = true;

      while (stack.length > 0) {
        const [cx, cy] = stack[stack.length - 1];
        const neighbors = getNeighbors(cx, cy, true);

        if (neighbors.length === 0) {
          stack.pop();
          continue;
        }

        const [nx, ny] = neighbors[Math.floor(Math.random() * neighbors.length)];
        removeWall(cx, cy, nx, ny);
        grid[ny][nx].visited = true;
        stack.push([nx, ny]);
      }
    }

    function generateMaze() {
      createGrid();
      generateSolutionPath();
      fillMaze();
    }
```

```javascript
    // === RENDERING ===

    function render() {
      ctx.fillStyle = COLORS.corridor;
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      drawMaze();
      drawTrail();
      drawPlayer();
      drawParticles();
    }

    function drawMaze() {
      ctx.strokeStyle = COLORS.wall;
      ctx.lineWidth = 1;

      for (let y = 0; y < ROWS; y++) {
        for (let x = 0; x < COLS; x++) {
          const px = x * CELL;
          const py = y * CELL;
          const cell = grid[y][x];

          ctx.beginPath();
          if (cell.walls.n) { ctx.moveTo(px, py + 0.5); ctx.lineTo(px + CELL, py + 0.5); }
          if (cell.walls.s) { ctx.moveTo(px, py + CELL - 0.5); ctx.lineTo(px + CELL, py + CELL - 0.5); }
          if (cell.walls.w) { ctx.moveTo(px + 0.5, py); ctx.lineTo(px + 0.5, py + CELL); }
          if (cell.walls.e) { ctx.moveTo(px + CELL - 0.5, py); ctx.lineTo(px + CELL - 0.5, py + CELL); }
          ctx.stroke();
        }
      }

      // Draw entry and exit openings with a highlight color
      ctx.fillStyle = COLORS.wallHighlight;
      const entryY = solutionPath[0][1];
      ctx.fillRect(0, entryY * CELL + 1, 2, CELL - 2);
      const exitY = solutionPath[solutionPath.length - 1][1];
      ctx.fillRect(COLS * CELL - 2, exitY * CELL + 1, 2, CELL - 2);
    }

    function drawPlayer() {
      if (gameState !== 'playing') return;
      const px = player.x * CELL + 2;
      const py = player.y * CELL + 2;
      ctx.fillStyle = COLORS.player;
      ctx.fillRect(px, py, CELL - 4, CELL - 4);
    }

    function drawTrail() {
      if (!trailEnabled || gameState !== 'playing') return;
      ctx.fillStyle = COLORS.trail;
      for (const [tx, ty] of player.trail) {
        ctx.fillRect(tx * CELL + 1, ty * CELL + 1, CELL - 2, CELL - 2);
      }
    }

    function drawParticles() {
      for (const p of particles) {
        ctx.globalAlpha = p.life;
        ctx.fillStyle = p.color;
        ctx.fillRect(Math.floor(p.x), Math.floor(p.y), Math.ceil(p.size), Math.ceil(p.size));
      }
      ctx.globalAlpha = 1;
    }
```

```javascript
    // === PLAYER & INPUT ===

    function canMove(x, y, dir) {
      // Block moving back out the entrance
      if (dir === 'w' && x === 0) return false;
      // Block east movement off the grid (exit handled separately in movePlayer)
      if (dir === 'e' && x === COLS - 1) return false;
      if (dir === 'n') return !grid[y][x].walls.n && y > 0;
      if (dir === 's') return !grid[y][x].walls.s && y < ROWS - 1;
      if (dir === 'w') return !grid[y][x].walls.w;
      if (dir === 'e') return !grid[y][x].walls.e;
      return false;
    }

    function movePlayer(dir) {
      if (gameState !== 'playing') return;

      // Check for exit before normal movement
      if (player.x === COLS - 1 && dir === 'e' && !grid[player.y][player.x].walls.e) {
        winGame();
        return;
      }

      if (!canMove(player.x, player.y, dir)) return;

      const dx = dir === 'e' ? 1 : dir === 'w' ? -1 : 0;
      const dy = dir === 's' ? 1 : dir === 'n' ? -1 : 0;
      const nx = player.x + dx;
      const ny = player.y + dy;

      player.trail.push([player.x, player.y]);
      player.x = nx;
      player.y = ny;

      // Spawn movement sparkles
      spawnParticles(player.x * CELL + CELL / 2, player.y * CELL + CELL / 2, 3, 'move');

      // Update checkpoint
      updateCheckpoint();

      // Check dead end
      if (isDeadEnd()) {
        showDeadEnd();
      }

      render();
    }

    document.addEventListener('keydown', (e) => {
      const keyMap = {
        ArrowUp: 'n', ArrowDown: 's', ArrowLeft: 'w', ArrowRight: 'e',
        w: 'n', s: 's', a: 'w', d: 'e'
      };
      const dir = keyMap[e.key];
      if (dir) {
        e.preventDefault();
        movePlayer(dir);
      }
    });
```

- [ ] **Step 1: Add checkpoint logic**

```javascript
    // === CHECKPOINT ===

    function buildSolutionSet() {
      // Map from "x,y" -> index in solutionPath for O(1) lookup
      player.solutionMap = new Map();
      for (let i = 0; i < solutionPath.length; i++) {
        const key = solutionPath[i][0] + ',' + solutionPath[i][1];
        // Only store the first occurrence (path is self-avoiding, so each cell appears once)
        if (!player.solutionMap.has(key)) {
          player.solutionMap.set(key, i);
        }
      }
    }

    function updateCheckpoint() {
      if (!checkpointEnabled) return;
      const key = player.x + ',' + player.y;
      if (player.solutionMap.has(key)) {
        const idx = player.solutionMap.get(key);
        if (idx >= player.lastCheckpointIndex) {
          player.lastCheckpointIndex = idx;
        }
      }
    }
```

```javascript
    // === DEAD END ===

    function isDeadEnd() {
      // Spec-aligned: dead end = all immediate passable neighbors are already visited.
      // The player cannot move to any unvisited cell directly from where they stand.
      const visited = new Set(player.trail.map(t => t[0] + ',' + t[1]));
      visited.add(player.x + ',' + player.y);

      const directions = [
        ['n', 0, -1], ['s', 0, 1], ['w', -1, 0], ['e', 1, 0]
      ];
      for (const [dir, dx, dy] of directions) {
        if (canMove(player.x, player.y, dir)) {
          const key = (player.x + dx) + ',' + (player.y + dy);
          if (!visited.has(key)) return false;
        }
      }
      return true;
    }

    function showDeadEnd() {
      gameState = 'deadEnd';
      const resumeBtn = document.getElementById('resume-btn');
      resumeBtn.style.display = checkpointEnabled ? 'inline-block' : 'none';
      document.getElementById('dead-end-screen').classList.remove('hidden');
      document.getElementById('help-btn').classList.add('hidden');
    }

    function restart() {
      gameState = 'playing';
      player.x = solutionPath[0][0];
      player.y = solutionPath[0][1];
      player.trail = [];
      player.lastCheckpointIndex = 0;
      document.getElementById('dead-end-screen').classList.add('hidden');
      document.getElementById('help-btn').classList.remove('hidden');
      render();
    }

    function resumeFromCheckpoint() {
      gameState = 'playing';
      const cp = solutionPath[player.lastCheckpointIndex];
      player.x = cp[0];
      player.y = cp[1];

      // Clear trail from checkpoint onward: find the LAST occurrence of the
      // checkpoint cell in the trail (player may have passed through it multiple times)
      const cpKey = cp[0] + ',' + cp[1];
      let cpTrailIdx = -1;
      for (let i = player.trail.length - 1; i >= 0; i--) {
        if (player.trail[i][0] + ',' + player.trail[i][1] === cpKey) {
          cpTrailIdx = i;
          break;
        }
      }
      if (cpTrailIdx >= 0) {
        player.trail = player.trail.slice(0, cpTrailIdx + 1);
      }
      // If checkpoint cell not in trail (player started there), clear entire trail

      spawnParticles(player.x * CELL + CELL / 2, player.y * CELL + CELL / 2, 10, 'teleport');

      document.getElementById('dead-end-screen').classList.add('hidden');
      document.getElementById('help-btn').classList.remove('hidden');
      render();
    }

    // Wire up buttons
    document.getElementById('restart-btn').addEventListener('click', restart);
    document.getElementById('resume-btn').addEventListener('click', resumeFromCheckpoint);
    document.getElementById('help-btn').addEventListener('click', () => {
      if (gameState === 'playing') showDeadEnd();
    });
```

```javascript
    // === PARTICLE SYSTEM ===

    function spawnParticles(cx, cy, count, type) {
      for (let i = 0; i < count; i++) {
        const angle = Math.random() * Math.PI * 2;
        const speed = type === 'win' ? (1 + Math.random() * 3) : (0.5 + Math.random() * 1.5);
        particles.push({
          x: cx,
          y: cy,
          vx: Math.cos(angle) * speed,
          vy: Math.sin(angle) * speed,
          life: 1.0,
          color: COLORS.sparkle[Math.floor(Math.random() * COLORS.sparkle.length)],
          size: type === 'win' ? (1 + Math.random() * 2) : (0.5 + Math.random())
        });
      }
    }

    function updateParticles(dt) {
      for (let i = particles.length - 1; i >= 0; i--) {
        const p = particles[i];
        p.x += p.vx * dt;
        p.y += p.vy * dt;
        p.life -= dt * 1.5;
        p.vx *= 0.98;
        p.vy *= 0.98;
        if (p.life <= 0) particles.splice(i, 1);
      }
    }
```
